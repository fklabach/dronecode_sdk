add_library(dronecode_sdk_manualcontrol ${PLUGIN_LIBRARY_TYPE}
    manualcontrol.cpp
    manualcontrol_impl.cpp
)

target_link_libraries(dronecode_sdk_manualcontrol
    dronecode_sdk
)

set_target_properties(dronecode_sdk_manualcontrol
    PROPERTIES COMPILE_FLAGS ${warnings}
)

install(FILES
    include/plugins/manualcontrol/manualcontrol.h
    DESTINATION ${dronecode_sdk_install_include_dir}
)

install(TARGETS dronecode_sdk_manualcontrol
    #EXPORT dronecode_sdk-targets
    DESTINATION ${dronecode_sdk_install_lib_dir}
)

target_include_directories(dronecode_sdk_manualcontrol
PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
)
